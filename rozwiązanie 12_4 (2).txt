SELECT A.SUMMARY,A.DESCRIPTION, B.NAME STATUS FROM issues A
JOIN issueslist B on B.ID = A.ISSUESLIST_ID;

SELECT I.SUMMARY, I.DESCRIPTION, U.LASTNAME,U.FIRSTNAME FROM issues I
JOIN users U on U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.ID,U.FIRSTNAME,U.LASTNAME, COUNT(*) LICZBA_ZADAN FROM issues I
JOIN users U on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID;
